## basic button:

```javascript
import React from 'react';
import Button from './components/Button';

function App() {
	return (
		<div className='App'>
			<Button size='small' bgColor='purple'>
				BUTTON
			</Button>
			<Button size='medium' bgColor='red'>
				BUTTON
			</Button>
			<Button size='big' bgColor='blue'>
				BUTTON
			</Button>
		</div>
	);
}

export default App;


```
